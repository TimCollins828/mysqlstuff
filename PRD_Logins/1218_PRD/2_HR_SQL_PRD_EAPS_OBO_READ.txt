--CREATE LOGIN if not exists WASHDC\HR_SQL_PRD_EAPS_OBO_READ
------------------------------------------------------------
USE [master]
GO
if not exists(select * from sys.server_principals where name = 'WASHDC\HR_SQL_PRD_EAPS_OBO_READ')
Begin
CREATE LOGIN [WASHDC\HR_SQL_PRD_EAPS_OBO_READ] FROM WINDOWS WITH DEFAULT_DATABASE=[EAPS], DEFAULT_LANGUAGE=[us_english]
END
GO
----------------------------------------------------------------
--Create the user [WASHDC\HR_SQL_PRD_EAPS_OBO_READ] in EAPS
----------------------------------------------------------------
Use [EAPS]
go
if exists(select * from sys.database_principals where name = 'WASHDC\HR_SQL_PRD_EAPS_OBO_READ')
Begin
Drop user [WASHDC\HR_SQL_PRD_EAPS_OBO_READ] 
END
Create user [WASHDC\HR_SQL_PRD_EAPS_OBO_READ] for Login [WASHDC\HR_SQL_PRD_EAPS_OBO_READ] with default_schema = dbo;
Grant connect to [WASHDC\HR_SQL_PRD_EAPS_OBO_READ]


GO
---------------------------------------------------------------- 
--Create Role [PRD_EAPS_OBO_SELECT_ONLY] in EAPS
---------------------------------------------------------------
Use EAPS
go
if exists (select 1 from sys.database_principals where name='PRD_EAPS_OBO_SELECT_ONLY' and Type = 'R')
Begin
ALTER ROLE PRD_EAPS_OBO_SELECT_ONLY DROP MEMBER [WASHDC\HR_SQL_PRD_EAPS_OBO_READ]
DROP ROLE PRD_EAPS_OBO_SELECT_ONLY
END
Create ROLE [PRD_EAPS_OBO_SELECT_ONLY];
GO
EXEC sp_addrolemember 'PRD_EAPS_OBO_SELECT_ONLY', 'WASHDC\HR_SQL_PRD_EAPS_OBO_READ'
go
